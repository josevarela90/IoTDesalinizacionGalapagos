state = None
ID = None
stateAnt = None


def setup():
  global rgb, espnow_0, pin1, F, tStart, state, ID, stateAnt

  M5.begin()
  rgb = RGB()
  rgb.set_brightness(0)
  espnow_0 = M5ESPNow(0)
  ID = 3
  espnow_0.set_add_peer('08F9E0D269F4', ID, 0, False)
  pin1 = Pin(1, mode=Pin.IN)
  stateAnt = pin1.value()


def loop():
  global rgb, espnow_0, pin1, F, tStart, state, ID, stateAnt
  F = 0
  tStart = time.ticks_ms()
  while (time.ticks_ms()) - tStart < 2000:
    state = pin1.value()
    if state and not stateAnt:
      F = F + 1
      stateAnt = 1
    if stateAnt and not state:
      stateAnt = 0
  F = (F / 4.8) / 2
  espnow_0.send_data(ID, int(F))
  F
  time.sleep(7)

    setup()

def entero_a_binario_8bits(numero):
    # Crear una lista para almacenar los bits
    bits = []

    # Realizar la conversión a binario manualmente
    while numero > 0:
        bits.append(str(numero % 2))  # Agregar el bit correspondiente
        numero //= 2

    # Asegurarse de que la lista tiene exactamente 8 bits
    while len(bits) < 8:
        bits.append('0')  # Agregar ceros para completar 8 bits

    # Invertir la lista y convertirla a una cadena
    return ''.join(bits[::-1])
